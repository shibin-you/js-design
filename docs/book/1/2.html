# 原型链

  • 创建对象有哪几种方法
var obj = {
  name: 'shibin'
}
var obj = new Object({
  name: 'shibin'
})

var Obj = function () {
  this.name = 'shibin'
}
var obj = new Obj()

var Obj = {
  name: 'shibin'
}
var obj = Object.create(Obj)
